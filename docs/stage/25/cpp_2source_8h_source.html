368 inline std::vector<uint16_t> Source::GetUniverses()
369 {
370  // This uses a guessing algorithm with a while loop to avoid race conditions.
371  std::vector<uint16_t> universes;
372  size_t size_guess = 4u;
373  size_t num_universes = 0u;
374 
375  do
376  {
377  universes.resize(size_guess);
378  num_universes = sacn_source_get_universes(handle_, universes.data(), universes.size());
379  size_guess = num_universes + 4u;
380  } while (num_universes > universes.size());
381 
382  universes.resize(num_universes);
383  return universes;
384 }

426 inline std::vector<etcpal::IpAddr> Source::GetUnicastDestinations(uint16_t universe)
427 {
428  // This uses a guessing algorithm with a while loop to avoid race conditions.
429  std::vector<EtcPalIpAddr> destinations;
430  size_t size_guess = 4u;
431  size_t num_destinations = 0u;
432 
433  do
434  {
435  destinations.resize(size_guess);
436  num_destinations = sacn_source_get_unicast_destinations(handle_, universe, destinations.data(), destinations.size());
437  size_guess = num_destinations + 4u;
438  } while (num_destinations > destinations.size());
439 
440  destinations.resize(num_destinations);
441 
442  // Convert vector<EtcPalIpAddr> to vector<etcpal::IpAddr>.
443  std::vector<etcpal::IpAddr> result;
444  if (!destinations.empty())
445  {
446  result.reserve(destinations.size());
447  std::transform(destinations.begin(), destinations.end(), std::back_inserter(result),
448  [](const EtcPalIpAddr& dest) { return etcpal::IpAddr(dest); });
449  }
450 
451  return result;
452 }
