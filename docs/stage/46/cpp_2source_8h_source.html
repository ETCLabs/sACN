400 inline std::vector<uint16_t> Source::GetUniverses()
401 {
402  // This uses a guessing algorithm with a while loop to avoid race conditions.
403  std::vector<uint16_t> universes;
404  size_t size_guess = 4u;
405  size_t num_universes = 0u;
406 
407  do
408  {
409  universes.resize(size_guess);
410  num_universes = sacn_source_get_universes(handle_, universes.data(), universes.size());
411  size_guess = num_universes + 4u;
412  } while (num_universes > universes.size());
413 
414  universes.resize(num_universes);
415  return universes;
416 }

457 inline std::vector<etcpal::IpAddr> Source::GetUnicastDestinations(uint16_t universe)
458 {
459  // This uses a guessing algorithm with a while loop to avoid race conditions.
460  std::vector<EtcPalIpAddr> destinations;
461  size_t size_guess = 4u;
462  size_t num_destinations = 0u;
463 
464  do
465  {
466  destinations.resize(size_guess);
467  num_destinations =
468  sacn_source_get_unicast_destinations(handle_, universe, destinations.data(), destinations.size());
469  size_guess = num_destinations + 4u;
470  } while (num_destinations > destinations.size());
471 
472  destinations.resize(num_destinations);
473 
474  // Convert vector<EtcPalIpAddr> to vector<etcpal::IpAddr>.
475  std::vector<etcpal::IpAddr> result;
476  if (!destinations.empty())
477  {
478  result.reserve(destinations.size());
479  std::transform(destinations.begin(), destinations.end(), std::back_inserter(result),
480  [](const EtcPalIpAddr& dest) { return etcpal::IpAddr(dest); });
481  }
482 
483  return result;
484 }
