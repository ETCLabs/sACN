416 inline std::vector<SacnMcastInterface> MergeReceiver::GetNetworkInterfaces()
417 {
418  // This uses a guessing algorithm with a while loop to avoid race conditions.
419  std::vector<SacnMcastInterface> netints;
420  size_t size_guess = 4u;
421  size_t num_netints = 0u;
422 
423  do
424  {
425  netints.resize(size_guess);
426  num_netints = sacn_merge_receiver_get_network_interfaces(handle_, netints.data(), netints.size());
427  size_guess = num_netints + 4u;
428  } while (num_netints > netints.size());
429 
430  netints.resize(num_netints);
431  return netints;
432 }
