469 inline std::vector<SacnMcastInterface> Receiver::GetNetworkInterfaces()
470 {
471  // This uses a guessing algorithm with a while loop to avoid race conditions.
472  std::vector<SacnMcastInterface> netints;
473  size_t size_guess = 4u;
474  size_t num_netints = 0u;
475 
476  do
477  {
478  netints.resize(size_guess);
479  num_netints = sacn_receiver_get_network_interfaces(handle_, netints.data(), netints.size());
480  size_guess = num_netints + 4u;
481  } while (num_netints > netints.size());
482 
483  netints.resize(num_netints);
484  return netints;
485 }
