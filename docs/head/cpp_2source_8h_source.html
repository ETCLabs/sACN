410 inline std::vector<uint16_t> Source::GetUniverses()
411 {
412  // This uses a guessing algorithm with a while loop to avoid race conditions.
413  std::vector<uint16_t> universes;
414  size_t size_guess = 4u;
415  size_t num_universes = 0u;
416 
417  do
418  {
419  universes.resize(size_guess);
420  num_universes = sacn_source_get_universes(handle_.value(), universes.data(), universes.size());
421  size_guess = num_universes + 4u;
422  } while (num_universes > universes.size());
423 
424  universes.resize(num_universes);
425  return universes;
426 }

468 inline std::vector<etcpal::IpAddr> Source::GetUnicastDestinations(uint16_t universe)
469 {
470  // This uses a guessing algorithm with a while loop to avoid race conditions.
471  std::vector<EtcPalIpAddr> destinations;
472  size_t size_guess = 4u;
473  size_t num_destinations = 0u;
474 
475  do
476  {
477  destinations.resize(size_guess);
478  num_destinations =
479  sacn_source_get_unicast_destinations(handle_.value(), universe, destinations.data(), destinations.size());
480  size_guess = num_destinations + 4u;
481  } while (num_destinations > destinations.size());
482 
483  destinations.resize(num_destinations);
484 
485  // Convert vector<EtcPalIpAddr> to vector<etcpal::IpAddr>.
486  std::vector<etcpal::IpAddr> result;
487  if (!destinations.empty())
488  {
489  result.reserve(destinations.size());
490  std::transform(destinations.begin(), destinations.end(), std::back_inserter(result),
491  [](const EtcPalIpAddr& dest) { return etcpal::IpAddr(dest); });
492  }
493 
494  return result;
495 }
